## Example Request

export const quotasExamples = [{
  title: "Python",
  language: "python",
  source: `import requests

headers = {
  "Authorization": "Bearer SYNTHETIC_API_KEY"
}

response = requests.post(
  "https://api.synthetic.new/v2/search",
  headers=headers,
  data={ "query": "docs for the python requests library" }
)

print(response.json())`,
}, {
  title: "TypeScript",
  language: "typescript",
  source: `const response = await fetch("https://api.synthetic.new/v2/search", {
  method: "POST",
  headers: {
    "Authorization": "Bearer SYNTHETIC_API_KEY",
    "Content-Type": "application/json",
  },
  body: JSON.stringify({
    query: "mdn docs for fetch",
  }),
});

const response = await response.json();
console.log(response.results);`,
}, {
  title: "curl",
  language: "bash",
  source: `curl https://api.synthetic.new/v2/search \\
  -H "Authorization: Bearer $\{SYNTHETIC_API_KEY\}" \\
  -H "Content-Type: application/json" \\
  -d '{ "query": "docs for curl" }'`,
}];
